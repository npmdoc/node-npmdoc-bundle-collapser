  • description and source-code
    bundle-collapser = function (src) {
    var rows = unpack(src);
    var p = pack({ raw: true });
    rows.forEach(function (row) {
        row.source = replace(row.source, row.deps);
        row.deps = {};
        p.write(row);
    });
    p.end();
    return p;
}
